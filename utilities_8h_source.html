248 /// Unroll an loop up to 4 iterations, where iterator is the identifier
249 /// used in each operation (e.g "i"), number_of_iterations is a constant which
250 /// specifies the number of times to perform the operation and "operation" is
251 /// the statement to execute for each iteration of the loop (e.g data[i] = v).
252 #define MATHFU_UNROLLED_LOOP(iterator, number_of_iterations, operation) \
253  { \
254  const int iterator = 0; \
255  { operation; } \
256  if ((number_of_iterations) > 1) { \
257  const int iterator = 1; \
258  { operation; } \
259  if ((number_of_iterations) > 2) { \
260  const int iterator = 2; \
261  { operation; } \
262  if ((number_of_iterations) > 3) { \
263  const int iterator = 3; \
264  { operation; } \
265  if ((number_of_iterations) > 4) { \
266  for (int iterator = 4; iterator < (number_of_iterations); \

268  operation; \

271  } \
272  } \

296 /// @brief Linearly interpolate between range_start and range_end, based on
297 /// percent.
298 /// @anchor mathfu_Lerp
299 ///
300 /// @param range_start Start of the range.
301 /// @param range_end End of the range.
302 /// @param percent Value between 0.0 and 1.0 used to interpolate between
303 /// range_start and range_end. Where a value of 0.0 results in a return
304 /// value of range_start and 1.0 results in a return value of range_end.
305 /// @return Value between range_start and range_end.
306 ///
307 /// @tparam T Type of the range to interpolate over.
308 /// @tparam T2 Type of the value used to perform interpolation
309 /// (e.g float or double).
310 template <class T, class T2>
311 T Lerp(const T &range_start, const T &range_end, const T2 &percent) {
312  const T2 one_minus_percent = static_cast<T2>(1.0) - percent;
313  return range_start * one_minus_percent + range_end * percent;
314 }

432 /// @addtogroup mathfu_allocator
433 ///
434 /// If you use MathFu with SIMD (SSE in particular), you need to have all
435 /// your allocations be 16-byte aligned (which isn't the case with the default
436 /// allocators on most platforms except OS X).
437 ///
438 /// You can either use simd_allocator, which solves the problem for
439 /// any STL containers, but not for manual dynamic allocations or the
440 /// new/delete override MATHFU_DEFINE_GLOBAL_SIMD_AWARE_NEW_DELETE will
441 /// solve it for all allocations, at the cost of MATHFU_ALIGNMENT bytes per
442 /// allocation.
443 
444 /// @addtogroup mathfu_allocator
445 /// @{
446 
447 /// @def MATHFU_ALIGNMENT
448 /// @brief Alignment (in bytes) of memory allocated by AllocateAligned.
449 ///
450 /// @see mathfu::AllocateAligned()
451 /// @see mathfu::simd_allocator
452 #define MATHFU_ALIGNMENT 16
453 
454 /// @brief Allocate an aligned block of memory.
455 /// @anchor mathfu_AllocateAligned
456 ///
457 /// This function allocates a block of memory aligned to MATHFU_ALIGNMENT
458 /// bytes.
459 ///
460 /// @param n Size of memory to allocate.
461 /// @return Pointer to aligned block of allocated memory or NULL if
462 /// allocation failed.
463 inline void *AllocateAligned(size_t n) {
464 #if defined(_MSC_VER) && _MSC_VER >= 1900 // MSVC 2015
465  return _aligned_malloc(n, MATHFU_ALIGNMENT);
466 #else
467  // We need to allocate extra bytes to guarantee alignment,
468  // and to store the pointer to the original buffer.
469  uint8_t *buf = reinterpret_cast<uint8_t *>(malloc(n + MATHFU_ALIGNMENT));
470  if (!buf) return NULL;
471  // Align to next higher multiple of MATHFU_ALIGNMENT.
472  uint8_t *aligned_buf = reinterpret_cast<uint8_t *>(
473  (reinterpret_cast<size_t>(buf) + MATHFU_ALIGNMENT) &
474  ~(MATHFU_ALIGNMENT - 1));
475  // Write out original buffer pointer before aligned buffer.
476  // The assert will fail if the allocator granularity is less than the pointer
477  // size, or if MATHFU_ALIGNMENT doesn't fit two pointers.
478  assert(static_cast<size_t>(aligned_buf - buf) > sizeof(void *));
479  *(reinterpret_cast<uint8_t **>(aligned_buf) - 1) = buf;
480  return aligned_buf;
481 #endif // defined(_MSC_VER) && _MSC_VER >= 1900 // MSVC 2015
482 }

484 /// @brief Deallocate a block of memory allocated with AllocateAligned().
485 /// @anchor mathfu_FreeAligned
486 ///
487 /// @param p Pointer to memory to deallocate.
488 inline void FreeAligned(void *p) {
489 #if defined(_MSC_VER) && _MSC_VER >= 1900 // MSVC 2015
490  _aligned_free(p);
491 #else
492  if (p == NULL) return;
493  free(*(reinterpret_cast<uint8_t **>(p) - 1));
494 #endif // defined(_MSC_VER) && _MSC_VER >= 1900 // MSVC 2015
495 }

497 /// @brief SIMD-safe memory allocator, for use with STL types like std::vector.
498 ///
499 /// For example:
500 /// <blockquote><code><pre>
501 /// std::vector<vec4, mathfu::simd_allocator<vec4>> myvector;
502 /// </pre></code></blockquote>
503 ///
504 /// @see MATHFU_DEFINE_GLOBAL_SIMD_AWARE_NEW_DELETE
505 /// @tparam T type allocated by this object.
506 template <typename T>
507 class simd_allocator : public std::allocator<T> {
508  public:
509  /// Size type.
510  typedef size_t size_type;
511  /// Pointer of type T.
512  typedef T *pointer;
513  /// Const pointer of type T.
514  typedef const T *const_pointer;

516  /// Constructs a simd_allocator.
517  simd_allocator() throw() : std::allocator<T>() {}
518  /// @brief Constructs and copies a simd_allocator.
519  ///
520  /// @param a Allocator to copy.
521  simd_allocator(const simd_allocator &a) throw() : std::allocator<T>(a) {}
522  /// @brief Constructs and copies a simd_allocator.
523  ///
524  /// @param a Allocator to copy.
525  /// @tparam U type of the object allocated by the allocator to copy.
526  template <class U>
527  simd_allocator(const simd_allocator<U> &a) throw() : std::allocator<T>(a) {}
528  /// @brief Destructs a simd_allocator.
